import React, { useState, useEffect } from 'react';
import {
  StyleSheet,
  Text,
  View,
  FlatList,
  Image,
  TouchableOpacity,
  ActivityIndicator,
  StatusBar,
} from 'react-native';
import { SafeAreaProvider, SafeAreaView } from 'react-native-safe-area-context';
import LinearGradient from 'react-native-linear-gradient';
import { Shadow } from 'react-native-shadow-2';
import { 
  Home as HomeIcon, 
  Search, 
  Image as CommunityIcon, 
  ShoppingBag, 
  MoreHorizontal 
} from 'lucide-react-native';
import { Tokens } from './src/theme/theme'; // Paths resolve directly to your central layout definitions tokens file

const MASTER_LIMIT = 198;

const TOTAL_SIDE_PADDING = Tokens.layout.paddingHorizontal * 2;
const DYNAMIC_FEED_CARD_WIDTH = (Tokens.layout.width - TOTAL_SIDE_PADDING - Tokens.gaps.small) / 2;

const OutfitFeed = ({ navigation }) => {
  const [allProducts, setAllProducts] = useState([]);
  const [activeCategories, setActiveCategories] = useState(['All']); 
  const [loading, setLoading] = useState(false);

  useEffect(() => {
    fetchMasterData();
  }, []);

  const fetchMasterData = () => {
    setLoading(true);
    const url = `https://dummyjson.com/products?limit=${MASTER_LIMIT}`;

    fetch(url)
      .then(response => response.json())
      .then(json => {
        if (json && json.products) {
          setAllProducts(json.products);
        }
      })
      .catch(error => console.error("Fetch failure:", error))
      .finally(() => setLoading(false));
  };

  const toggleCategory = (category) => {
    if (category === 'All') {
      setActiveCategories(['All']); 
      return;
    }

    setActiveCategories(prev => {
      const cleanPrev = prev.filter(c => c !== 'All'); 

      if (cleanPrev.includes(category)) {
        const updated = cleanPrev.filter(c => c !== category);
        return updated.length === 0 ? ['All'] : updated;
      } else {
        return [...cleanPrev, category];
      }
    });
  };

  const displayedProducts = allProducts.filter(product => {
    if (activeCategories.includes('All')) return true;
    return activeCategories.includes(product.category);
  });

  const openProductDisplay = item => {
    if (navigation) {
      navigation.push('ProductDisplay', { product: item });
    }
  };

  const renderCircleItem = ({ item }) => (
    <TouchableOpacity
      style={styles.circleItemContainer}
      activeOpacity={0.85}
      onPress={() => openProductDisplay(item)}
    >
      <LinearGradient
        colors={['#FDABAC', '#FDEABF']}
        start={{ x: 0.01, y: 0.5 }}
        end={{ x: 0.99, y: 0.5 }}
        style={styles.circleGradientRing}
      >
        <View style={styles.circleWrapperInner}>
          <Image
            source={{ uri: item.images?.[0] || item.thumbnail }}
            style={styles.avatarImage}
            resizeMode="cover"
          />
        </View>
      </LinearGradient>
    </TouchableOpacity>
  );

  const renderHeader = () => (
    <View style={styles.headerContainer}>
      
      {/* Category Grid Frame Layout Container */}
      <View style={styles.chipGridContainer}>
        
        {/* Row 1 */}
        <View style={styles.chipRow}>
          
          {/* Chip 1: Everyday Fit */}
          <TouchableOpacity 
            onPress={() => toggleCategory('All')} 
            activeOpacity={0.85}
            style={{ width: Tokens.scaleAsset(142) }}
          >
            {activeCategories.includes('All') ? (
              <Shadow
                distance={10}                          // Expanded distance to give the fading room to disperse smoothly
                startColor="rgba(251, 147, 96, 0.64)" // Lower opacity at origin point creates a softer neon glow
                endColor="rgba(0, 0, 0, 0)"            // Completely fades into dark theme canvas color space
                offset={[0, 0]}                        // [Sidewise=0, Downward=3] position mapping coordinates
                paintInside={false}                    // Essential configuration: prevents shadow from overlapping inner view bounds
                containerStyle={styles.shadowContainerFluid}
                style={styles.shadowStyleFluid}
              >
                <LinearGradient
                  colors={['#FBB59E', '#F8876C', '#F16646', '#F98F7A']}
                  start={{ x: 0, y: 0 }}
                  end={{ x: 1, y: 0 }}
                  style={styles.gradientOuterBorder}
                >
                  <View style={styles.chipInnerContentContainerActive}>
                    <Text style={styles.chipTextActive}>👟 Everyday Fit</Text>
                  </View>
                </LinearGradient>
              </Shadow>
            ) : (
              <View style={styles.inactiveChipContainer}>
                <Text style={styles.chipTextInactive}>👟 Everyday Fit</Text>
              </View>
            )}
          </TouchableOpacity>

          {/* Chip 2: Work Fit */}
          <TouchableOpacity 
            onPress={() => toggleCategory('beauty')} 
            activeOpacity={0.85}
            style={{ width: Tokens.scaleAsset(112) }}
          >
            {activeCategories.includes('beauty') ? (
              <Shadow
                distance={10}
                startColor="rgba(248, 135, 108, 0.32)"
                endColor="rgba(0, 0, 0, 0)"
                offset={[0, 0]}
                paintInside={false}
                containerStyle={styles.shadowContainerFluid}
                style={styles.shadowStyleFluid}
              >
                <LinearGradient
                  colors={['#FBB59E', '#F8876C', '#F16646', '#F98F7A']}
                  start={{ x: 0, y: 0 }}
                  end={{ x: 1, y: 0 }}
                  style={styles.gradientOuterBorder}
                >
                  <View style={styles.chipInnerContentContainerActive}>
                    <Text style={styles.chipTextActive}>💼 Work Fit</Text>
                  </View>
                </LinearGradient>
              </Shadow>
            ) : (
              <View style={styles.inactiveChipContainer}>
                <Text style={styles.chipTextInactive}>💼 Work Fit</Text>
              </View>
            )}
          </TouchableOpacity>
        </View>

        {/* Row 2 */}
        <View style={styles.chipRow}>
          
          {/* Chip 3: Winter Vacation Fit */}
          <TouchableOpacity 
            onPress={() => toggleCategory('laptops')} 
            activeOpacity={0.85}
            style={{ width: Tokens.scaleAsset(188) }}
          >
            {activeCategories.includes('laptops') ? (
              <Shadow
                distance={8}
                startColor="rgba(248, 135, 108, 0.32)"
                endColor="rgba(0, 0, 0, 0)"
                offset={[0, 1.7]}
                paintInside={false}
                containerStyle={styles.shadowContainerFluid}
                style={styles.shadowStyleFluid}
              >
                <LinearGradient
                  colors={['#FBB59E', '#F8876C', '#F16646', '#F98F7A']}
                  start={{ x: 0, y: 0 }}
                  end={{ x: 1, y: 0 }}
                  style={styles.gradientOuterBorder}
                >
                  <View style={styles.chipInnerContentContainerActive}>
                    <Text style={styles.chipTextActive}>❄️ Winter Vacation Fit</Text>
                  </View>
                </LinearGradient>
              </Shadow>
            ) : (
              <View style={styles.inactiveChipContainer}>
                <Text style={styles.chipTextInactive}>❄️ Winter Vacation Fit</Text>
              </View>
            )}
          </TouchableOpacity>

          {/* Chip 4: Date Night Fit */}
          <TouchableOpacity 
            onPress={() => toggleCategory('mens-watches')} 
            activeOpacity={0.85}
            style={{ width: Tokens.scaleAsset(152) }}
          >
            {activeCategories.includes('mens-watches') ? (
              <Shadow
                distance={10}
                startColor="rgba(248, 135, 108, 0.32)"
                endColor="rgba(0, 0, 0, 0)"
                offset={[0, 0]}
                paintInside={false}
                containerStyle={styles.shadowContainerFluid}
                style={styles.shadowStyleFluid}
              >
                <LinearGradient
                  colors={['#FBB59E', '#F8876C', '#F16646', '#F98F7A']}
                  start={{ x: 0, y: 0 }}
                  end={{ x: 1, y: 0 }}
                  style={styles.gradientOuterBorder}
                >
                  <View style={styles.chipInnerContentContainerActive}>
                    <Text style={styles.chipTextActive}>💃 Date Night Fit</Text>
                  </View>
                </LinearGradient>
              </Shadow>
            ) : (
              <View style={styles.inactiveChipContainer}>
                <Text style={styles.chipTextInactive}>💃 Date Night Fit</Text>
              </View>
            )}
          </TouchableOpacity>
        </View>

        {/* Row 3 */}
        <View style={styles.chipRow}>
          
          {/* Chip 5: Summer Vacation Fit */}
          <TouchableOpacity 
            onPress={() => toggleCategory('womens-dresses')} 
            activeOpacity={0.85}
            style={{ width: Tokens.scaleAsset(200) }}
          >
            {activeCategories.includes('womens-dresses') ? (
              <Shadow
                distance={8}
                startColor="rgba(248, 135, 108, 0.32)"
                endColor="rgba(0, 0, 0, 0)"
                offset={[0, 1.7 ]}
                
                paintInside={false}
                containerStyle={styles.shadowContainerFluid}
                style={styles.shadowStyleFluid}
              >
                <LinearGradient
                  colors={['#FBB59E', '#F8876C', '#F16646', '#F98F7A']}
                  start={{ x: 0, y: 0 }}
                  end={{ x: 1, y: 0 }}
                  style={styles.gradientOuterBorder}
                >
                  <View style={styles.chipInnerContentContainerActive}>
                    <Text style={styles.chipTextActive}>🩴 Summer Vacation Fit</Text>
                  </View>
                </LinearGradient>
              </Shadow>
            ) : (
              <View style={styles.inactiveChipContainer}>
                <Text style={styles.chipTextInactive}>🩴 Summer Vacation Fit</Text>
              </View>
            )}
          </TouchableOpacity>

          {/* Chip 6: Night Out Fit */}
          <TouchableOpacity 
            onPress={() => toggleCategory('home-decoration')} 
            activeOpacity={0.85}
            style={{ width: Tokens.scaleAsset(144) }}
          >
            {activeCategories.includes('home-decoration') ? (
              <Shadow
                distance={14}
                startColor="rgba(248, 135, 108, 0.32)"
                endColor="rgba(0, 0, 0, 0)"
                offset={[0, 3]}
                paintInside={false}
                containerStyle={styles.shadowContainerFluid}
                style={styles.shadowStyleFluid}
              >
                <LinearGradient
                  colors={['#FBB59E', '#F8876C', '#F16646', '#F98F7A']}
                  start={{ x: 0, y: 0 }}
                  end={{ x: 1, y: 0 }}
                  style={styles.gradientOuterBorder}
                >
                  <View style={styles.chipInnerContentContainerActive}>
                    <Text style={styles.chipTextActive}>✨ Night Out Fit</Text>
                  </View>
                </LinearGradient>
              </Shadow>
            ) : (
              <View style={styles.inactiveChipContainer}>
                <Text style={styles.chipTextInactive}>✨ Night Out Fit</Text>
              </View>
            )}
          </TouchableOpacity>
        </View>

      </View>

      {/* Line Divider */}
      <View style={styles.lineDivider} />

      {/* Stories Avatar Carousel Component */}
      <FlatList
        data={displayedProducts.slice(0, 10)} 
        renderItem={renderCircleItem}
        keyExtractor={item => `circle-${item.id}`}
        horizontal
        showsHorizontalScrollIndicator={false}
        contentContainerStyle={styles.circleCarouselPadding}
      />
    </View>
  );

  const renderGridItem = ({ item }) => (
    <TouchableOpacity
      style={styles.clothingCard}
      activeOpacity={0.9}
      onPress={() => openProductDisplay(item)}
    >
      <Image
        source={{ uri: item.images?.[0] || item.thumbnail }}
        style={styles.clothingCardImage}
        resizeMode="cover"
      />
    </TouchableOpacity>
  );

  return (
    <SafeAreaProvider>
      <LinearGradient
        colors={['#0F0F0F', '#0D0D0D']}
        start={{ x: 0.44, y: 0 }}
        end={{ x: 0.54, y: 0.98 }}
        style={styles.screenContainer}
      >
        <StatusBar barStyle="light-content" backgroundColor="#0F0F0F" />
        
        <SafeAreaView style={styles.mainContainer} edges={['top', 'left', 'right']}>
          <FlatList
            data={displayedProducts}
            renderItem={renderGridItem} 
            keyExtractor={item => `grid-${item.id}`}
            numColumns={2}
            columnWrapperStyle={styles.gridRowGap}
            ListHeaderComponent={renderHeader}
            contentContainerStyle={styles.scrollContainerPadding}
            showsVerticalScrollIndicator={false}
          />
        </SafeAreaView>

        {/* Global Navigation Footer Structure */}
        <View style={styles.footerContainer}>
          <TouchableOpacity style={styles.footerTabFrame} activeOpacity={0.7}>
            <HomeIcon size={Tokens.scaleAsset(20)} color="#F8876C" strokeWidth={2} />
            <Text style={[styles.footerTabText, { color: '#F8876C' }]}>Home</Text>
          </TouchableOpacity>

          <TouchableOpacity style={styles.footerTabFrame} activeOpacity={0.7}>
            <Search size={Tokens.scaleAsset(20)} color="#B3B3B3" strokeWidth={2} />
            <Text style={styles.footerTabText}>Explore</Text>
          </TouchableOpacity>

          <TouchableOpacity style={[styles.footerTabFrame, { width: Tokens.scaleAsset(104) }]} activeOpacity={0.7}>
            <CommunityIcon size={Tokens.scaleAsset(20)} color="#B3B3B3" strokeWidth={2} />
            <Text style={styles.footerTabText}>Community Space</Text>
          </TouchableOpacity>

          <TouchableOpacity style={styles.footerTabFrame} activeOpacity={0.7}>
            <ShoppingBag size={Tokens.scaleAsset(20)} color="#B3B3B3" strokeWidth={2} />
            <Text style={styles.footerTabText}>Cart</Text>
          </TouchableOpacity>

          <TouchableOpacity style={styles.footerTabFrame} activeOpacity={0.7}>
            <MoreHorizontal size={Tokens.scaleAsset(20)} color="#B3B3B3" strokeWidth={2} />
            <Text style={styles.footerTabText}>More</Text>
          </TouchableOpacity>
        </View>

      </LinearGradient>
    </SafeAreaProvider>
  );
};
